# Runtime dependencies for the Hello! Project Wiki chatbot.
#
# These are the *direct* dependencies. Transitive dependencies are
# resolved by pip automatically.
#
# Build-time dependencies (mwxml, mwparserfromhell, tiktoken) are
# NOT listed here because they're only needed by build_index.py and
# build_embeddings.py, which are run locally before deployment. The
# runtime just loads the pre-built indexes.

# Anthropic SDK (used for MiniMax via the Anthropic-compat endpoint,
# and for real Anthropic).
anthropic>=0.85.0,<1.0

# OpenAI SDK (used for real OpenAI and Ollama via OpenAI-compat).
openai>=1.0.0,<3.0

# ChromaDB (vector storage). Pin to 1.5.9 exactly because:
# - The local index was built with chromadb 1.5.9. Different 1.x
#   versions have different SQLite schemas and Rust API internals.
#   A version mismatch causes "pool timed out" and
#   "'RustBindingsAPI' has no attribute 'bindings'" errors.
# - 1.5.9 is the latest 1.x release as of 2026-05.
chromadb==1.5.9

# sentence-transformers is needed at runtime by tool_semantic_search
# in chat.py. It dynamically loads SentenceTransformer to embed the
# user's query. The model we use is BAAI/bge-small-en-v1.5.
sentence-transformers>=3.0,<5.0

# numpy is a transitive dep but it's used in some template formatters
# for chunk manipulation, so pin it explicitly.
numpy>=1.24,<3.0
